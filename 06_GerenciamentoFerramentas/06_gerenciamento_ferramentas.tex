	\begin{quote}
	``A project is a temporary endeavor with a defined beginning and end (\textbf{usually time-constrained, and often constrained by funding or deliverables}), undertaken to meet unique goals and objectives [...].''

\begin{frame}
	\frametitle{Ditador benevolente (BD)}
	\begin{block}{O que é?}
    \begin{itemize}
    \item Autoridade final na tomada de decisões\vspace{0.1cm}
    \item Um juiz \textbf{aprovado pela comunidade}\vspace{0.1cm}
    \item Muitas vezes é o fundador do projeto

    \end{block}

    \item Desenvolvedor experiente\vspace{0.1cm}

    \item {\em Principal habilidade}: reconhecer \bred{boas soluções de projeto}\vspace{0.1cm}
    \item Via de regra, é óbvio quem {\bf deve} ser o BD   
    \end{itemize}
    \end{block}

\begin{frame}
	\frametitle{Democracia baseada em consenso}
	\begin{block}{Funcionamento básico}
    \begin{itemize}
    \item Decisão a ser tomada é apresentada publicamente\vspace{0.1cm}
    \item Se não houver objeções, consenso é atingido\vspace{0.1cm}
    \item Decisão pode ser implementada\vspace{0.1cm}
    % ``implementada'' em um sentido mais amplo. Não precisa ser uma decisão diretamente
    % ligada ao código. 
    \end{itemize}
    \end{block}
\pause
	\begin{block}{Complementarmente}
    \begin{itemize}
    \item Algumas decisões podem ser tomadas sem discussão\vspace{0.1cm}
      \begin{itemize}
      \item Por exemplo, a decisão de corrigir um \textit{bug}
      \end{itemize}
    \item Consequências podem ser {\bf desfeitas}\vspace{0.1cm}
    % A possibilidade de desfazer as consequências de uma má decisão, possível graças ao
    % sistema de controle de versões, agiliza o processo de tomada de decisão e evita
    % que muito tempo seja gasto propondo publicamente cursos de ação óbvios. 
    \item Se consenso não for atingido, é necessário \byellow{votar}  
    \end{itemize}
    \end{block}
\end{frame}

\begin{frame}
	\frametitle{Controle de versões}
    \begin{block}{Terminologia}
    \begin{itemize}
    \item Versão %  É uma instância de um sistema que é funcionalmente distinta, de alguma maneira, de outras instâncias de um sistema.
    \item Variante % Uma versão de um sistema que tem apenas pequenas diferenças com relação a outras instâncias (normalmente devido a diferenças no hardware/software alvo)?. Por exemplo, o Office para MacOS é uma variante do Office para Windows.
    \item \textit{Release} % É uma instância de um sistema estável e distribuída para os usuários fora da equipe de desenvolvimento e de testes. Versões que não são releases mas que estão disponíveis para (algumas) pessoas fora da equipe de desenvolvimento são chamadas de versões alpha, beta ou "de desenvolvimento". 
    \end{itemize}
    \end{block}
    
    Esquemas de numeração registram a evolução do sistema.
    \pause
    \begin{block}{Numeração no Linux}
    \bred{{\bf A}}.\bgreen{{\bf B}}.\byellow{{\bf C}}\tblue{{\bf [.D]}}
    \begin{itemize}
    \item \bred{{\bf A}} -- Versão do núcleo (três mudanças, em 1994,1996 e 2011)
    \item \bgreen{{\bf B}} -- Revisão importante 
    \item \byellow{{\bf C}} -- mudanças menores (drivers e funcionalidades individuais) 
    \item \tblue{{\bf [.D]}} -- correções de bugs e atualizações de segurança 
    \end{itemize}
    \end{block}
\end{frame}

\begin{frame}
	\frametitle{\textit{Branching} no Git e no SVN}
    \begin{itemize}
    \item No Git:
    \begin{description}
    \item[] {\texttt{git branch \textit{$<$nome\_do\_branch$>$}}}\vspace{0.1cm}
    % git branch sem parâmetros diz apenas quais branches o repositório tem.
    \item[] {\texttt{git checkout \textit{$<$nome\_do\_branch$>$}}} \vspace{0.1cm}
    % git checkout muda o branch atual. Além disso, atualiza os arquivos na árvore de trabalho de acordo com os do branch.
    \item[] {\texttt{git merge \textit{$<$nome\_do\_branch$>$}}}
    \end{description}
    \item No SVN:
    \begin{description}
    \item[] {\texttt{svn copy \textit{$<$url\_1$>$} \textit{$<$url\_2$>$}}}
    \item[] {\texttt{svn merge \textit{$<$url\_1$>$} \textit{$<$url\_2$>$} \textit{$<copia\_trabalho>$}}}
    % O comando merge tem um nome esquisito. Ele não simplesmente combina dois branches como o nome sugere. Ao invés disso,
    % compara duas árvores de repositório (como o comando diff) e aplica as diferenças a uma cópia de trabalho. O comando
    % merge também pode ser usado para combinar branches. 
    \end{description}
    \end{itemize}
\end{frame}

\begin{frame}
	\frametitle{Para começar um novo projeto}
    \begin{itemize}
    \item Escolha um nome \vspace{0.1cm}
    \item Defina objetivos de longo prazo\vspace{0.1cm}
    \item Deixe claro que o projeto é livre\vspace{0.1cm}
    \item Elenque os requisitos e funcionalidades\vspace{0.1cm}
    \item Torne o programa baixável em formato fonte \vspace{0.1cm}
    \item Estabeleça acesso a controle de versões e rastreador de bugs\vspace{0.1cm}
      % Vários sítios podem ser usados para este fim. Atualmente, o Github é bastante
      % popular por combinar repositório com controle de versões baseado no Git, bug
      % tracking e rede social em um só sítio. 
    \item Estabeleça canais de comunicação (lista, canal de IRC)\vspace{0.1cm}
    % De forma geral, meios para que pessoas possam se comunicar e cooperar.
    \item Escolha a licença\vspace{0.1cm}
    \item \textbf{Anuncie}!
    % Em http://www.freecode.com
    \end{itemize}
\end{frame}

    \item Crie uma página Web
    \begin{itemize}
